# plugin-type: plugin-exec
"""Zerodha Kite — V3 user-`.txt` exec plugin.

Places / cancels / modifies orders and reports positions + holdings +
margins for one alias. Shares the alias's :class:`KiteSession` with the
matching ``kite_broker_data.txt`` via
:mod:`opentrader.connectors_v2.session_registry` so OAuth happens
exactly once per account.

SEBI static-IP compliance: when ``proxy_url`` is provided in credentials,
order placement routes through that SOCKS5 proxy. Historical / realtime
data bypasses the proxy (no static-IP requirement for read-only APIs).

The :class:`KiteSession` class body MUST be byte-identical between this
file and ``kite_broker_data.txt`` per the V3 §R2 Decision-3=3d contract.
"""

# Kite Connect access tokens expire at 06:00 IST every trading day.
# https://kite.trade/docs/connect/v3/user/#user
IST = timezone(timedelta(hours=5, minutes=30))
_KITE_TOKEN_EXPIRY_TIME_IST = dtime(6, 0)


def _next_kite_token_expiry(now: datetime | None = None) -> datetime:
    """Return the next 06:00 IST after ``now``."""
    now = (now or datetime.now(IST)).astimezone(IST)
    expiry_today = now.replace(
        hour=_KITE_TOKEN_EXPIRY_TIME_IST.hour,
        minute=_KITE_TOKEN_EXPIRY_TIME_IST.minute,
        second=0, microsecond=0,
    )
    if now < expiry_today:
        return expiry_today
    return expiry_today + timedelta(days=1)

# ── Local redirect server for OAuth ────────────────────────────────────
_REDIRECT_PORT = 5678
_REDIRECT_PATH = "/callback"


class _TokenCaptureHandler(http.server.BaseHTTPRequestHandler):
    """HTTP handler that captures the Kite OAuth request_token."""

    captured_token: str = ""

    def do_GET(self):                                  # noqa: N802
        parsed = urlparse(self.path)
        if parsed.path == _REDIRECT_PATH:
            token = parse_qs(parsed.query).get("request_token", [""])[0]
            _TokenCaptureHandler.captured_token = token
            self.send_response(200)
            self.send_header("Content-Type", "text/html")
            self.end_headers()
            body = (
                b"<h2>Login Successful!</h2><p>Return to OpenTrader-Pro.</p>"
                if token else b"<h2>Login Failed</h2><p>No token received.</p>"
            )
            self.wfile.write(b"<html><body>" + body + b"</body></html>")
        else:
            self.send_response(404)
            self.end_headers()

    def log_message(self, format, *args):              # noqa: A002
        pass                                            # silence log noise


# ══════════════════════════════════════════════════════════════════════
# KiteSession — one per alias. BYTE-IDENTICAL with kite_broker_data.txt.
# Both sides construct via session_registry.get_or_create_session(
#   namespace="zerodha", alias=…, factory=KiteSession) so a single
# OAuth login covers both data and exec for the same alias.
# ══════════════════════════════════════════════════════════════════════
class KiteSession:
    """Owns a pair of ``KiteConnect`` instances (direct + proxied) for one alias.

    Never construct directly — always go through
    ``session_registry.get_or_create_session(namespace="zerodha", alias=…,
    factory=KiteSession)`` so the data + exec ``.txt`` files share one
    OAuth flow per alias.
    """

    def __init__(self, alias: str) -> None:
        self.alias = alias
        self._api_key: str = ""
        self._api_secret: str = ""
        self._access_token: str = ""
        self._authenticated: bool = False
        # Wall-clock when the current access token becomes invalid.
        # Kite mints daily-expiry tokens; populated at successful login.
        self._expires_at: datetime | None = None

        self._kite: Any | None = None             # direct
        self._kite_exec: Any | None = None        # proxied

        # Symbol/token caches
        self._instruments: dict[str, list[dict]] = {}
        self._token_cache: dict[str, int] = {}
        self._reverse_token_cache: dict[int, str] = {}
        # symbol → exchange; order-dependent when a trading symbol exists on
        # multiple exchanges (RELIANCE is on both NSE + BSE). Unsafe for tick
        # routing — prefer ``_token_exch_cache`` which is keyed on the
        # globally-unique instrument token.
        self._exch_lookup: dict[str, str] = {}
        # token → exchange; populated at the same time as reverse_token_cache
        # so ``_on_raw_ticks`` can disambiguate without guessing.
        self._token_exch_cache: dict[int, str] = {}
        self._lot_size_cache: dict[str, int] = {}

        # Serialize concurrent REST calls (KiteConnect uses requests.Session).
        # Public so data / exec plugins can wrap their own direct kite calls.
        self.api_lock = threading.Lock()

        # Order-update push channel. KiteTicker (created by the data side)
        # exposes ``on_order_update`` postbacks; the exec side registers a
        # parser via ``register_order_update_handler`` and the data side's
        # ticker dispatches into it through ``dispatch_order_update``.
        # Storing it here means data + exec stay decoupled — neither imports
        # the other — while sharing the single ticker WS connection.
        self._order_update_handler: Callable[[dict], None] | None = None

    # ── Basic predicates ─────────────────────────────────────────────
    @property
    def is_available(self) -> bool:
        return KITE_AVAILABLE

    @property
    def is_configured(self) -> bool:
        return bool(self._api_key and self._api_secret)

    @property
    def is_authenticated(self) -> bool:
        return self._authenticated

    @property
    def expires_at(self) -> datetime | None:
        """When the current access token becomes invalid (None if no session)."""
        return self._expires_at

    @property
    def kite(self) -> Any:
        return self._kite

    @property
    def kite_exec(self) -> Any:
        """Proxied instance, or direct if no proxy configured."""
        return self._kite_exec if self._kite_exec is not None else self._kite

    @property
    def has_proxy(self) -> bool:
        """True iff a proxied :class:`KiteConnect` instance is installed."""
        return self._kite_exec is not None

    # ── Configuration ────────────────────────────────────────────────
    def configure(self, api_key: str, api_secret: str) -> bool:
        """Create (or reuse) the direct KiteConnect instance for this alias."""
        if not KITE_AVAILABLE:
            return False
        if (
            self._kite is not None
            and self._api_key == api_key
            and self._api_secret == api_secret
        ):
            return True
        self._api_key = api_key
        self._api_secret = api_secret
        self._kite = KiteConnect(api_key=api_key)
        if self._authenticated and self._access_token:
            self._kite.set_access_token(self._access_token)
        logger.info(
            "KiteSession[%s] configured (api_key=%s***)", self.alias, api_key[:4]
        )
        return True

    def configure_exec_proxy(self, proxy_url: str | None) -> None:
        """Create (or clear) the proxied exec KiteConnect instance."""
        if not KITE_AVAILABLE or not self._api_key:
            return

        if not proxy_url:
            self._kite_exec = None
            logger.info("KiteSession[%s] exec proxy cleared.", self.alias)
            return

        if proxy_url.startswith("socks5://"):
            proxy_url = proxy_url.replace("socks5://", "socks5h://", 1)

        proxies = {"http": proxy_url, "https": proxy_url}
        self._kite_exec = KiteConnect(api_key=self._api_key, proxies=proxies)
        if hasattr(self._kite_exec, "reqsession"):
            self._kite_exec.reqsession.proxies.update(proxies)
        if self._authenticated and self._access_token:
            self._kite_exec.set_access_token(self._access_token)
        logger.info(
            "KiteSession[%s] exec proxy: %s", self.alias, proxy_url
        )

    # ── OAuth ────────────────────────────────────────────────────────
    def login(self, timeout: int = 120) -> tuple[bool, str]:
        """Browser-based OAuth login. Blocking — call from a worker thread."""
        if not KITE_AVAILABLE:
            return False, "kiteconnect not installed."
        if not self.is_configured:
            return False, "Not configured. Provide api_key + api_secret."

        try:
            _TokenCaptureHandler.captured_token = ""
            server = http.server.HTTPServer(
                ("127.0.0.1", _REDIRECT_PORT), _TokenCaptureHandler
            )
            server.timeout = 2
            server_thread = threading.Thread(
                target=self._run_redirect_server,
                args=(server, timeout), daemon=True,
                name=f"KiteRedirect-{self.alias}",
            )
            server_thread.start()

            webbrowser.open(self._kite.login_url())
            start = time.time()
            while time.time() - start < timeout:
                if _TokenCaptureHandler.captured_token:
                    break
                time.sleep(0.5)
            server_thread.join(timeout=5)

            request_token = _TokenCaptureHandler.captured_token
            if not request_token:
                return False, "Login timed out — no request token received."

            data = self._kite.generate_session(
                request_token, api_secret=self._api_secret
            )
            self._set_access_token(data["access_token"])
            self._authenticated = True
            self._expires_at = _next_kite_token_expiry()
            threading.Thread(
                target=self._load_instruments, daemon=True,
                name=f"KiteInstruments-{self.alias}",
            ).start()
            user = data.get("user_name", data.get("user_id", ""))
            logger.info(
                "KiteSession[%s] login OK — user %s, token expires %s",
                self.alias, user, self._expires_at.isoformat(timespec="minutes"),
            )
            return True, f"Login successful. Welcome, {user}!"
        except Exception as exc:
            self._authenticated = False
            logger.error("KiteSession[%s] login failed: %s", self.alias, exc)
            return False, f"Login failed: {exc}"

    def login_with_token(self, request_token: str) -> tuple[bool, str]:
        """Manual fallback when the browser flow isn't available."""
        if not self.is_configured:
            return False, "Not configured."
        try:
            data = self._kite.generate_session(
                request_token, api_secret=self._api_secret
            )
            self._set_access_token(data["access_token"])
            self._authenticated = True
            self._expires_at = _next_kite_token_expiry()
            self._load_instruments()
            return True, f"Login successful. User: {data.get('user_name', '')}"
        except Exception as exc:
            return False, f"Login failed: {exc}"

    def logout(self) -> None:
        if self._kite and self._authenticated:
            try:
                self._kite.invalidate_access_token(self._access_token)
            except Exception as exc:
                logger.warning(
                    "KiteSession[%s] logout warn: %s", self.alias, exc
                )
        self._authenticated = False
        self._access_token = ""
        self._expires_at = None
        if self._kite_exec:
            try:
                self._kite_exec.set_access_token("")
            except Exception as exc:
                logger.debug(
                    "KiteSession[%s] proxy clear-token failed: %s",
                    self.alias, exc,
                )
        logger.info("KiteSession[%s] logged out.", self.alias)

    # ── Instruments cache ────────────────────────────────────────────
    def _load_instruments(self) -> None:
        if not self._authenticated or not self._kite:
            return
        for exch in ("NSE", "BSE", "MCX", "NFO", "CDS"):
            try:
                with self.api_lock:
                    instruments = self._kite.instruments(exch)
                self._instruments[exch] = instruments
                for inst in instruments:
                    sym = inst["tradingsymbol"]
                    token = inst["instrument_token"]
                    key = f"{exch}:{sym}"
                    self._token_cache[key] = token
                    self._reverse_token_cache[token] = sym
                    self._token_exch_cache[token] = exch
                    self._exch_lookup[sym] = exch
                    self._lot_size_cache[key] = inst.get("lot_size", 1)
                logger.info(
                    "KiteSession[%s] cached %d %s instruments",
                    self.alias, len(instruments), exch,
                )
            except Exception as exc:
                logger.warning(
                    "KiteSession[%s] failed %s instruments: %s",
                    self.alias, exch, exc,
                )

    def get_instruments(self, exchange: str = "NSE") -> list[dict]:
        return self._instruments.get(exchange, [])

    def resolve_token(
        self,
        symbol: str,
        exchange: str = "NSE",
        *,
        wait_timeout: float = 15.0,
    ) -> int | None:
        """Look up ``symbol`` on ``exchange``; return instrument token or None."""
        key = f"{exchange}:{symbol}"
        token = self._token_cache.get(key)
        if token is not None:
            return token
        if wait_timeout > 0 and not self._instruments.get(exchange):
            deadline = time.time() + wait_timeout
            while time.time() < deadline and not self._instruments.get(exchange):
                time.sleep(0.1)
            token = self._token_cache.get(key)
            if token is not None:
                return token
        for inst in self._instruments.get(exchange, []):
            if inst["tradingsymbol"] == symbol:
                t = inst["instrument_token"]
                self._token_cache[key] = t
                self._reverse_token_cache[t] = symbol
                self._token_exch_cache[t] = exchange
                return t
        return None

    def find_exchange(self, symbol: str) -> str | None:
        exch = self._exch_lookup.get(symbol)
        if exch:
            return exch
        for e in ("NSE", "NFO", "MCX", "CDS", "BSE"):
            if f"{e}:{symbol}" in self._token_cache:
                return e
        return None

    def get_lot_size(self, symbol: str, exchange: str) -> int:
        return self._lot_size_cache.get(f"{exchange}:{symbol}", 1)

    def get_symbol_from_token(self, token: int) -> str | None:
        return self._reverse_token_cache.get(token)

    def get_exchange_from_token(self, token: int) -> str | None:
        """Return the exchange a token was loaded from."""
        return self._token_exch_cache.get(token)

    def search_instruments(
        self, query: str, exchange: str = "NSE", limit: int = 20
    ) -> list[dict]:
        q = query.upper()
        out: list[dict] = []
        for inst in self._instruments.get(exchange, []):
            if (
                q in inst.get("tradingsymbol", "").upper()
                or q in inst.get("name", "").upper()
            ):
                out.append({
                    "symbol": inst["tradingsymbol"],
                    "name": inst.get("name", ""),
                    "exchange": exchange,
                    "type": inst.get("instrument_type", "EQ"),
                    "lot_size": inst.get("lot_size", 1),
                    "tick_size": inst.get("tick_size", 0.05),
                })
                if len(out) >= limit:
                    break
        return out

    # ── Ticker ───────────────────────────────────────────────────────
    def create_ticker(self) -> Any | None:
        """KiteTicker uses direct connection; live data never needs static IP."""
        if not KITE_AVAILABLE or not self._authenticated:
            return None
        return KiteTicker(self._api_key, self._access_token)

    # ── Order-update push channel (V3 §F7) ───────────────────────────
    def register_order_update_handler(
        self, handler: Callable[[dict], None] | None
    ) -> None:
        """Register a parser for KiteTicker ``on_order_update`` postbacks."""
        self._order_update_handler = handler

    def dispatch_order_update(self, data: dict) -> None:
        """Forward a Kite postback dict to the registered handler."""
        handler = self._order_update_handler
        if handler is None:
            return
        try:
            handler(data)
        except Exception as exc:
            logger.exception(
                "KiteSession[%s] order-update handler raised: %s",
                self.alias, exc,
            )

    # ── Internals ────────────────────────────────────────────────────
    def _set_access_token(self, token: str) -> None:
        self._access_token = token
        if self._kite:
            self._kite.set_access_token(token)
        if self._kite_exec:
            self._kite_exec.set_access_token(token)

    def _dispose(self) -> None:
        """Drop KiteConnect instances. Called by session_registry when the
        last user releases this session."""
        self._kite = None
        self._kite_exec = None
        self._instruments.clear()
        self._token_cache.clear()
        self._reverse_token_cache.clear()
        self._token_exch_cache.clear()
        self._exch_lookup.clear()
        self._lot_size_cache.clear()
        self._authenticated = False
        self._access_token = ""

    @staticmethod
    def _run_redirect_server(
        server: http.server.HTTPServer, timeout: int
    ) -> None:
        start = time.time()
        while time.time() - start < timeout:
            if _TokenCaptureHandler.captured_token:
                break
            server.handle_request()
        try:
            server.server_close()
        except Exception as exc:
            logger.debug("OAuth redirect server close failed: %s", exc)

# Errors — V3 §F9 broker-error taxonomy mapping.
# Inlined from former opentrader/connectors_v2/plugins/kite_broker/errors.py.
# ══════════════════════════════════════════════════════════════════════
_INSUFFICIENT_FUNDS_HINTS: tuple[str, ...] = (
    "insufficient funds", "margin shortfall", "available margin",
    "insufficient margin", "shortage of funds",
)
_RATE_LIMIT_HINTS: tuple[str, ...] = (
    "too many requests", "rate limit", "throttle", "429",
)


def classify_kite_exception(exc: BaseException) -> BrokerError:
    """Translate a vendor exception into the V3 broker-error taxonomy."""
    msg = str(exc) if exc.args else exc.__class__.__name__
    low = msg.lower()

    try:
        from kiteconnect import exceptions as kex  # type: ignore[import-not-found]
    except ImportError:
        kex = None  # type: ignore[assignment]

    if kex is not None:
        if isinstance(exc, kex.TokenException):
            return AuthError(msg)
        if isinstance(exc, kex.PermissionException):
            return AuthError(msg)
        if isinstance(exc, kex.NetworkException):
            return TransientError(msg)
        if isinstance(exc, kex.InputException):
            return InvalidOrderError(msg)
        if isinstance(exc, kex.OrderException):
            if any(h in low for h in _INSUFFICIENT_FUNDS_HINTS):
                return InsufficientFundsError(msg)
            if any(h in low for h in _RATE_LIMIT_HINTS):
                return RateLimitError(msg)
            return BrokerRejectError(msg)
        if isinstance(exc, (kex.DataException, kex.GeneralException)):
            return BrokerRejectError(msg)

    if isinstance(exc, (TimeoutError, ConnectionError)):
        return TransientError(msg)
    if isinstance(exc, OSError):
        return TransientError(msg)

    if any(h in low for h in _RATE_LIMIT_HINTS):
        return RateLimitError(msg)
    if any(h in low for h in _INSUFFICIENT_FUNDS_HINTS):
        return InsufficientFundsError(msg)

    return BrokerRejectError(msg)

def _is_option(symbol: str) -> bool:
    """Retail heuristic: NFO symbols ending ``CE`` / ``PE`` are options."""
    s = (symbol or "").upper()
    return s.endswith("CE") or s.endswith("PE")


def _round2(x: float) -> float:
    """Money-friendly 2-decimal rounding."""
    return round(x, 2)


def _fees_zero() -> Fees:
    return Fees.zero(currency=_CCY)


def compute_zerodha_fees(trade: TradeFill) -> Fees:
    """Compute :class:`Fees` for one Zerodha fill."""
    qty = float(trade.quantity or 0.0)
    price = float(trade.price or 0.0)
    if qty <= 0 or price <= 0.0:
        return _fees_zero()

    turnover = qty * price
    side_u = (trade.side or "").upper()
    seg_u = (trade.segment or "").upper()
    prod_u = (trade.product_type or "").upper()

    if seg_u in ("NSE_EQ", "BSE_EQ"):
        if prod_u == "CNC":
            return _fees_equity_delivery(turnover, side_u)
        return _fees_equity_intraday(turnover, side_u)

    if seg_u == "NFO":
        if _is_option(trade.symbol):
            return _fees_options(turnover, side_u)
        return _fees_futures(turnover, side_u)

    if seg_u == "MCX":
        return _fees_mcx(turnover, side_u)

    return _fees_zero()


def _fees_equity_delivery(turnover: float, side: str) -> Fees:
    brokerage = 0.0
    stt = turnover * _STT_EQ_DELIVERY
    exch = turnover * _TXN_NSE_EQ
    sebi = turnover * _SEBI_RATE
    stamp = turnover * _STAMP_DELIVERY_BUY if side == "BUY" else 0.0
    gst = (brokerage + exch + sebi) * _GST_RATE
    return Fees(
        brokerage=_round2(brokerage),
        exchange_charges=_round2(exch),
        taxes={
            "stt":          _round2(stt),
            "gst":          _round2(gst),
            "sebi_charges": _round2(sebi),
            "stamp_duty":   _round2(stamp),
        },
        currency=_CCY,
    )

# MCX non-agri DST flip — module-level helpers
# ══════════════════════════════════════════════════════════════════════
#
# MCX bullion / metals / energy commodities reference US futures
# markets for pricing, so their close time shifts when the US enters /
# exits Daylight Saving Time:
#
#   * Summer (US DST active, mid-Mar–early-Nov): 09:00–23:55 IST
#   * Winter (US standard time, early-Nov–mid-Mar): 09:00–23:30 IST
#
# Pre-Phase-6 the plugin hard-coded 09:00–23:55 year-round, which:
#   * dropped ~25 min of valid trading every winter day (orders during
#     23:30–23:55 silently rejected by broker as post-close), and
#   * forced winter users to write a ``market_hours.json`` override.
#
# These helpers compute the flip from the system tzdata
# (America/New_York) so the rule stays correct across future DST
# legislation changes without source edits. Mirrors the data-side
# plugin (see ``kite_broker_data.txt``); kept in sync because the exec
# gate (``SignalRouter._market_open``) consults this exec plugin and
# the data gate (``RealtimeManager.is_exchange_tradeable``) consults
# the data plugin — both must agree on session boundaries.

def _today_in_ist(_clock=datetime.now) -> date:
    """Return today's date in IST. The ``_clock`` injection point lets
    tests pass a fixed ``datetime`` provider for deterministic
    DST-boundary assertions; production calls fall through to
    ``datetime.now``."""
    return _clock(tz=ZoneInfo("Asia/Kolkata")).date()


def _is_us_dst_active(on: date) -> bool:
    """True when New York is observing DST on date ``on``. Uses system
    tzdata (``zoneinfo.ZoneInfo``) — survives future US DST legislation
    changes without code edits. Pure."""
    ny = ZoneInfo("America/New_York")
    noon_ny = datetime.combine(on, dtime(12, 0), tzinfo=ny)
    return noon_ny.dst() != timedelta(0)


def _mcx_nonagri_close_time(today: date | None = None) -> dtime:
    """Resolve today's MCX non-agri close time:

    * 23:55 IST when US is in DST (summer, ~mid-Mar–early-Nov)
    * 23:30 IST otherwise (winter, early-Nov–mid-Mar)

    ``today`` defaults to ``_today_in_ist()``; tests pin it.
    """
    if today is None:
        today = _today_in_ist()
    return dtime(23, 55) if _is_us_dst_active(today) else dtime(23, 30)

    # ── Construction ─────────────────────────────────────────────────
    def __init__(self, alias: str) -> None:
        super().__init__(alias)
        self._session: KiteSession = session_registry.get_or_create_session(
            namespace=self.INFO.name, alias=alias, factory=KiteSession,
        )
        self._configured = False
        self._err_state: dict[str, tuple[str, float]] = {}
        self._proxy_fail_count: int = 0
        self._proxy_open_at: float | None = None
        self._monotonic = time.monotonic

    # ── Lifecycle ────────────────────────────────────────────────────
    def configure(
        self,
        credentials: dict[str, str],
        *,
        routing: SessionRouting | None = None,
    ) -> None:
        if not KITE_AVAILABLE:
            raise RuntimeError(
                "kiteconnect not installed. Run: pip install kiteconnect"
            )
        api_key = credentials.get("api_key", "")
        api_secret = credentials.get("api_secret", "")
        if not api_key or not api_secret:
            raise ValueError("api_key and api_secret are required")
        if routing is not None:
            proxy_url = routing.routed.proxy_url
        else:
            cred_proxy = (credentials.get("proxy_url") or "").strip() or None
            proxy_url = (
                _validate_exec_proxy_url(cred_proxy)
                if cred_proxy is not None
                else None
            )
        self._session.configure(api_key, api_secret)
        self._session.configure_exec_proxy(proxy_url)
        # Wire the order-update push channel (V3 §F7). The KiteTicker the
        # data side owns will dispatch postbacks into our parser via the
        # shared session — polling stays as fallback for outages.
        self._session.register_order_update_handler(self._on_kite_order_update)
        self._configured = True

    def authenticate(self) -> AuthResult:
        if not KITE_AVAILABLE:
            return AuthResult(ok=False, error="kiteconnect not installed")
        if not self._configured:
            return AuthResult(ok=False, error="not configured")
        if self._session.is_authenticated:
            self._authenticated = True
            return AuthResult(ok=True, expires_at=self._session.expires_at)
        ok, msg = self._session.login()
        if ok:
            self._authenticated = True
            return AuthResult(ok=True, expires_at=self._session.expires_at)
        return AuthResult(ok=False, error=msg)

    def authenticate_with_token(self, token: str) -> AuthResult:
        """Manual fallback when the OAuth redirect listener can't reach back."""
        if not KITE_AVAILABLE:
            return AuthResult(ok=False, error="kiteconnect not installed")
        if not self._configured:
            return AuthResult(ok=False, error="not configured")
        if not token or not token.strip():
            return AuthResult(ok=False, error="request_token is empty")
        ok, msg = self._session.login_with_token(token.strip())
        if ok:
            self._authenticated = True
            return AuthResult(ok=True, expires_at=self._session.expires_at)
        return AuthResult(ok=False, error=msg)

    def test_connection(self) -> bool:
        if not self._configured or not self._session.is_authenticated:
            return False
        try:
            self._session.kite_exec.profile()
            return True
        except Exception:
            return False

    @property
    def is_authenticated(self) -> bool:
        return self._session.is_authenticated

    def shutdown(self) -> None:
        self._session.register_order_update_handler(None)
        session_registry.release_session(self.INFO.name, self.alias)
        super().shutdown()

    def get_timezone(self, segment: str | None = None) -> ZoneInfo:
        """Zerodha covers Indian exchanges; all segments share IST."""
        return ZoneInfo("Asia/Kolkata")

    def get_holiday_calendar(
        self, segment: str | None = None
    ) -> tuple[date, ...]:
        """Return the NSE/BSE/NFO trading-holiday baseline (V3 §F5)."""
        del segment
        return NSE_2026

    def get_market_hours(
        self, segment: str | None = None
    ) -> tuple[MarketWindow, ...]:
        """Return Zerodha's regular trading session (V3 §F6).

        Per-segment Indian market sessions (all IST):

        * Equity (NSE/BSE cash + F&O): 09:15–15:30
        * Currency derivatives (CDS/BCD): **09:00–17:00**
          — pre-fix this was bundled with equity 09:15–15:30, which
          dropped live ticks in the 09:00–09:15 pre-open window AND
          15:30–17:00 post-equity-close window. The signal router's
          ``_market_open`` gate rejected legitimate currency orders
          during those windows; once fixed, the gate accepts them.
        * MCX commodities — three sub-sessions:
          - ``MCX_AGRI`` (Cotton, CPO, Kapas): 09:00–21:00
          - ``MCX_AGRI_EARLY`` (Cardamom, Mentha, Castor): 09:00–17:00
          - ``MCX_NONAGRI`` / plain ``MCX`` (Bullion, Metals, Energy):
            **date-aware** — 09:00–23:55 when US is in DST
            (~mid-Mar–early-Nov), 09:00–23:30 otherwise. Flip is
            driven by US commodity-market reference pricing; plugin
            reads system tzdata so the rule stays correct across
            future US DST legislation changes.

        Pre-fix every MCX_* segment returned 09:00–23:55 year-round, so
        agri orders could pass router gate but be rejected by broker
        at place_order time AND winter non-agri orders slipped through
        the 23:30–23:55 gap the broker had closed.
        """
        ist = ZoneInfo("Asia/Kolkata")
        equity              = MarketWindow(dtime( 9, 15), dtime(15, 30), tz=ist)
        currency            = MarketWindow(dtime( 9,  0), dtime(17,  0), tz=ist)
        # Non-agri close flips with US DST — see ``_mcx_nonagri_close_time``
        # at module level for the rationale.
        commodity_nonagri   = MarketWindow(
            dtime( 9,  0), _mcx_nonagri_close_time(), tz=ist,
        )
        commodity_agri      = MarketWindow(dtime( 9,  0), dtime(21,  0), tz=ist)
        commodity_agri_early= MarketWindow(dtime( 9,  0), dtime(17,  0), tz=ist)
        seg = (segment or "").upper()
        if seg in {"CDS", "BCD", "NSE_CDS", "BSE_CDS"}:
            return (currency,)
        if seg == "MCX_AGRI":
            return (commodity_agri,)
        if seg == "MCX_AGRI_EARLY":
            return (commodity_agri_early,)
        if seg.startswith("MCX"):
            return (commodity_nonagri,)
        if seg in {"NSE", "NSE_EQ", "BSE", "BSE_EQ", "NFO", "BFO"}:
            return (equity,)
        # No segment hint — return the broadest union an Indian-equity
        # exec plugin reasonably covers (equity + non-agri commodities).
        return (equity, commodity_nonagri)

    def calculate_fees(self, trade: TradeFill) -> Fees:
        """Delegate to the inlined India fee schedule."""
        return compute_zerodha_fees(trade)

    # ── Orders ───────────────────────────────────────────────────────
    def place_order(self, request: OrderRequest) -> OrderResponse:
        if request.account_id != self.account_id:
            return OrderResponse(ok=False, error="account mismatch")
        if not self._session.is_authenticated:
            return OrderResponse(ok=False, error="not authenticated")
        if request.quantity <= 0:
            return OrderResponse(ok=False, error="quantity must be positive")
        if (early := self._check_proxy_circuit()) is not None:
            return early

        kite = self._session.kite_exec
        order_type = ORDER_TYPE_MAP.get(request.order_type, "MARKET")
        product = PRODUCT_MAP.get(request.product, "CNC")
        exchange = request.exchange or "NSE"

        params: dict = {
            "variety": "regular",
            "tradingsymbol": request.symbol,
            "exchange": exchange,
            "transaction_type": "BUY" if request.side == "BUY" else "SELL",
            "quantity": request.quantity,
            "order_type": order_type,
            "product": product,
            "validity": "DAY",
            "price": request.price if order_type in ("LIMIT", "STOP_LIMIT") else None,
            "trigger_price": (
                request.trigger_price if order_type in ("STOP_LIMIT", "STOP") else None
            ),
            "tag": request.client_order_id,
        }
        if order_type in ("MARKET", "STOP"):
            params["market_protection"] = -1

        try:
            order_id = kite.place_order(**params)
        except Exception as exc:
            self._record_proxy_failure(exc)
            logger.error(
                "Zerodha[%s] order failed: %s", self.alias, exc
            )
            raise classify_kite_exception(exc) from exc
        self._record_proxy_success()
        logger.info(
            "Zerodha[%s] %s %s:%s qty=%d → %s",
            self.alias, params["transaction_type"],
            exchange, request.symbol, request.quantity, order_id,
        )
        return OrderResponse(ok=True, broker_order_id=str(order_id))

    def cancel_order(self, order_id: str) -> OrderResponse:
        if not self._session.is_authenticated:
            return OrderResponse(ok=False, error="not authenticated")
        if (early := self._check_proxy_circuit()) is not None:
            return early
        try:
            self._session.kite_exec.cancel_order(
                variety="regular", order_id=order_id
            )
        except Exception as exc:
            self._record_proxy_failure(exc)
            logger.error(
                "Zerodha[%s] cancel failed: %s", self.alias, exc
            )
            raise classify_kite_exception(exc) from exc
        self._record_proxy_success()
        return OrderResponse(ok=True, broker_order_id=order_id)

    def modify_order(
        self, order_id: str, changes: OrderModification
    ) -> OrderResponse:
        if not self._session.is_authenticated:
            return OrderResponse(ok=False, error="not authenticated")
        if (early := self._check_proxy_circuit()) is not None:
            return early
        try:
            self._session.kite_exec.modify_order(
                variety="regular",
                order_id=order_id,
                quantity=changes.quantity,
                price=changes.price,
                trigger_price=changes.trigger_price,
                order_type=changes.order_type,
            )
        except Exception as exc:
            self._record_proxy_failure(exc)
            logger.error(
                "Zerodha[%s] modify failed: %s", self.alias, exc
            )
            raise classify_kite_exception(exc) from exc
        self._record_proxy_success()
        return OrderResponse(ok=True, broker_order_id=order_id)

    # ── Proxy circuit breaker ────────────────────────────────────────
    def _is_proxy_active(self) -> bool:
        """True iff a proxied :class:`KiteConnect` was installed."""
        return self._session.has_proxy

    @staticmethod
    def _is_proxy_error(exc: BaseException) -> bool:
        """Heuristic: does this exception look proxy-induced?"""
        msg = str(exc)
        low = msg.lower()
        if "SOCKS" in msg:
            return True
        if "proxy" in low and ("refused" in low or "unreachable" in low):
            return True
        if "Max retries" in msg and "SOCKS" in msg:
            return True
        return False

    def _check_proxy_circuit(self) -> OrderResponse | None:
        """Return a fast-reject response when the circuit is open."""
        if self._proxy_open_at is None:
            return None
        if not self._is_proxy_active():
            self._proxy_open_at = None
            self._proxy_fail_count = 0
            return None
        elapsed = self._monotonic() - self._proxy_open_at
        remaining = self._PROXY_BREAKER_COOLDOWN_SECS - elapsed
        if remaining <= 0:
            self._proxy_open_at = None
            return None
        return OrderResponse(
            ok=False,
            error=(
                f"exec proxy unreachable; orders blocked for static-IP "
                f"compliance ({remaining:.0f}s until next probe) — fix "
                f"the proxy or clear proxy_url in Broker Manager"
            ),
        )

    def _record_proxy_failure(self, exc: BaseException) -> None:
        """Increment the proxy failure count when ``exc`` looks proxy-induced."""
        if not self._is_proxy_active() or not self._is_proxy_error(exc):
            return
        self._proxy_fail_count += 1
        if (
            self._proxy_fail_count >= self._PROXY_BREAKER_THRESHOLD
            and self._proxy_open_at is None
        ):
            self._proxy_open_at = self._monotonic()
            logger.warning(
                "Zerodha[%s] exec-proxy circuit OPEN after %d consecutive "
                "failures; orders blocked for %.0fs",
                self.alias,
                self._proxy_fail_count,
                self._PROXY_BREAKER_COOLDOWN_SECS,
            )

    def _record_proxy_success(self) -> None:
        """Any successful proxied call closes the breaker."""
        if self._proxy_fail_count == 0 and self._proxy_open_at is None:
            return
        if self._proxy_open_at is not None:
            logger.info(
                "Zerodha[%s] exec-proxy recovered — circuit CLOSED",
                self.alias,
            )
        self._proxy_fail_count = 0
        self._proxy_open_at = None

    # ── Error-log throttle (shared by all query methods) ─────────────
    @staticmethod
    def _error_signature(exc: BaseException) -> str:
        """Stable-ish dedup key: exception type + first line of message."""
        msg = str(exc).split("\n", 1)[0]
        return f"{type(exc).__name__}:{msg[:200]}"

    @staticmethod
    def _proxy_hint(exc: BaseException) -> str:
        """Return an actionable tail for SOCKS-proxy connection errors."""
        msg = str(exc)
        if "SOCKS" in msg and ("refused" in msg.lower() or "Max retries" in msg):
            return (
                " — SOCKS exec-proxy unreachable. Start the proxy, or "
                "clear proxy_url in Broker Manager if SEBI static-IP "
                "routing is not required. Subsequent identical failures "
                "will be suppressed."
            )
        return ""

    def _log_query_failure(self, method: str, exc: BaseException) -> None:
        """Log ``method`` failures with per-signature throttling."""
        self._record_proxy_failure(exc)
        sig = self._error_signature(exc)
        now = time.monotonic()
        last = self._err_state.get(method)
        is_new = (
            last is None
            or last[0] != sig
            or (now - last[1]) > self._THROTTLE_RESET_SECS
        )
        if is_new:
            self._err_state[method] = (sig, now)
            logger.error(
                "Zerodha[%s] %s() failed: %s%s",
                self.alias, method, exc, self._proxy_hint(exc),
            )
        else:
            logger.debug(
                "Zerodha[%s] %s() still failing (suppressed): %s",
                self.alias, method, exc,
            )

    def _reset_error_throttle(self, method: str) -> None:
        """Clear throttle state for ``method`` after a successful call."""
        self._err_state.pop(method, None)
        self._record_proxy_success()

    # ── Queries ──────────────────────────────────────────────────────
    def get_order_book(self) -> list[OrderRecord]:
        if not self._session.is_authenticated:
            return []
        try:
            rows = self._session.kite_exec.orders() or []
        except Exception as exc:
            self._log_query_failure("orders", exc)
            return []
        self._reset_error_throttle("orders")
        out: list[OrderRecord] = []
        for o in rows:
            placed = o.get("order_timestamp")
            if isinstance(placed, str):
                try:
                    placed = datetime.fromisoformat(placed).replace(tzinfo=IST)
                except ValueError:
                    placed = datetime.now(IST)
            elif hasattr(placed, "astimezone"):
                placed = placed.astimezone(IST)
            else:
                placed = datetime.now(IST)
            status = _map_order_status(o.get("status", ""))
            out.append(OrderRecord(
                broker_order_id=str(o["order_id"]),
                symbol=o["tradingsymbol"],
                exchange=o["exchange"],
                side=o["transaction_type"],
                quantity=int(o["quantity"]),
                filled_quantity=int(o.get("filled_quantity", 0)),
                order_type=o["order_type"],
                product=o["product"],
                status=status,
                price=float(o["price"]) if o.get("price") else None,
                trigger_price=(
                    float(o["trigger_price"]) if o.get("trigger_price") else None
                ),
                placed_at=placed,
                reason=o.get("status_message", ""),
            ))
        return out

    def get_positions(self) -> list[Position]:
        if not self._session.is_authenticated:
            return []
        try:
            positions = self._session.kite_exec.positions() or {}
        except Exception as exc:
            self._log_query_failure("positions", exc)
            return []
        self._reset_error_throttle("positions")
        out: list[Position] = []
        for p in positions.get("net", []):
            if p.get("quantity", 0) == 0:
                continue
            out.append(Position(
                symbol=p["tradingsymbol"],
                exchange=p["exchange"],
                quantity=int(p["quantity"]),
                average_price=float(p["average_price"]),
                pnl=float(p.get("pnl", 0.0)),
                product=p["product"],
            ))
        return out

    def get_holdings(self) -> list[Holding]:
        if not self._session.is_authenticated:
            return []
        try:
            rows = self._session.kite_exec.holdings() or []
        except Exception as exc:
            self._log_query_failure("holdings", exc)
            return []
        self._reset_error_throttle("holdings")
        return [
            Holding(
                symbol=h["tradingsymbol"],
                exchange=h["exchange"],
                quantity=int(h["quantity"]),
                average_price=float(h["average_price"]),
                last_price=float(h["last_price"]),
                pnl=float(h.get("pnl", 0.0)),
            )
            for h in rows
        ]

    # ── Order-update push channel (V3 §F7) ───────────────────────────
    def _on_kite_order_update(self, data: dict) -> None:
        """Parse a KiteTicker postback into an :class:`ExecutionReport`."""
        broker_order_id = str(data.get("order_id") or "").strip()
        if not broker_order_id:
            return
        kite_status = str(data.get("status") or "").upper()
        exec_type = _KITE_STATUS_TO_EXEC_TYPE.get(kite_status)
        if exec_type is None:
            return
        try:
            cumulative_qty = float(data.get("filled_quantity") or 0)
            avg_price = float(data.get("average_price") or 0)
        except (TypeError, ValueError):
            cumulative_qty = 0.0
            avg_price = 0.0
        ts_raw = data.get("order_timestamp") or data.get("exchange_timestamp")
        if isinstance(ts_raw, datetime):
            ts = ts_raw.astimezone(IST) if ts_raw.tzinfo else ts_raw.replace(tzinfo=IST)
        elif isinstance(ts_raw, str):
            try:
                ts = datetime.fromisoformat(ts_raw).replace(tzinfo=IST)
            except ValueError:
                ts = datetime.now(IST)
        else:
            ts = datetime.now(IST)
        report = ExecutionReport(
            order_id=str(data.get("tag") or ""),
            broker_order_id=broker_order_id,
            exec_type=exec_type,
            cumulative_qty=cumulative_qty,
            last_qty=cumulative_qty,
            last_price=avg_price,
            avg_price=avg_price,
            timestamp=ts,
            raw=dict(data),
        )
        self._emit_execution_report(report)

    def get_margins(self) -> Margins:
        if not self._session.is_authenticated:
            return Margins(available=0.0, used=0.0, total=0.0)
        try:
            m = self._session.kite_exec.margins("equity") or {}
        except Exception as exc:
            self._log_query_failure("margins", exc)
            raise
        self._reset_error_throttle("margins")
        avail = float(m.get("available", {}).get("cash", 0.0))
        used = float(m.get("utilised", {}).get("debits", 0.0))
        total = float(m.get("net", avail))
        return Margins(available=avail, used=used, total=total)


# ── Helpers ────────────────────────────────────────────────────────────
_ACCEPTED_PROXY_SCHEMES = ("socks5", "socks5h")


def _validate_exec_proxy_url(url: str) -> str:
    """Validate a SEBI static-IP exec proxy URL.

    Only ``socks5://`` and ``socks5h://`` schemes are accepted: HTTP/HTTPS
    proxies terminate the TCP connection and re-originate it from the
    proxy host, which means the broker sees the proxy's IP — fine for
    static-IP routing — but Kite's TLS pinning + connection-reuse model
    works far better with the SOCKS path.
    """
    parsed = urlparse(url)
    if parsed.scheme not in _ACCEPTED_PROXY_SCHEMES:
        raise ValueError(
            f"proxy_url scheme must be one of "
            f"{_ACCEPTED_PROXY_SCHEMES}; got {parsed.scheme!r} "
            f"(URL: {url!r})"
        )
    if not parsed.hostname:
        raise ValueError(
            f"proxy_url is missing host (e.g. socks5h://127.0.0.1:1080); "
            f"got {url!r}"
        )
    if parsed.port is None:
        raise ValueError(
            f"proxy_url is missing port (e.g. socks5h://127.0.0.1:1080); "
            f"got {url!r}"
        )
    return url

    "COMPLETE": "COMPLETE",
    "OPEN": "PENDING",
    "TRIGGER PENDING": "PENDING",
    "PENDING": "PENDING",
    "VALIDATION PENDING": "PENDING",

    "OPEN PENDING": "PENDING",
    "MODIFY PENDING": "PENDING",

}


# Kite native status → V3 ExecReportType (V3 §F7).
_KITE_STATUS_TO_EXEC_TYPE: dict[str, str] = {
    "COMPLETE": "FILL",
    "OPEN": "NEW",
    "TRIGGER PENDING": "NEW",
    "PUT ORDER REQ RECEIVED": "NEW",
    "OPEN PENDING": "NEW",
    "VALIDATION PENDING": "NEW",
    "MODIFY PENDING": "MODIFIED",
    "CANCELLED": "CANCELED",
    "REJECTED": "REJECTED",
}


def _map_order_status(native: str) -> str:
    """Translate Zerodha's order status to the v2 taxonomy."""
    return _STATUS_MAP.get(native.upper(), "PENDING")
