# plugin-type: indicator
"""Simple Moving Average indicator."""

import pandas as pd
import talib

from opentrader.indicators.base import BaseIndicator


class SMA(BaseIndicator):
    """Simple Moving Average — rolling mean of close prices."""

    name = "Simple Moving Average"
    short_name = "SMA"
    description = "Average of closing prices over a specified period."
    overlay_on_price = True

    def __init__(self) -> None:
        super().__init__()
        self.parameters = {"period": 20}
        self.add_line("sma", "SMA", color="#2962FF")

    def calculate(self, df: pd.DataFrame) -> None:
        period = self.parameters["period"]
        close = self._col(df, "close")
        self.lines["sma"].data = talib.SMA(close, timeperiod=period)
