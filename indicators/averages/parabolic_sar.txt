# plugin-type: indicator
import pandas as pd
import talib

from opentrader.indicators.base import BaseIndicator


class ParabolicSAR(BaseIndicator):
    """Parabolic SAR (Stop and Reverse)."""

    name = "Parabolic SAR"
    short_name = "PSAR"
    description = "Trend-following indicator that provides entry/exit points."
    overlay_on_price = True

    def __init__(self) -> None:
        super().__init__()
        self.parameters = {"acceleration": 0.02, "maximum": 0.2}
        self.add_line("sar", "SAR", style="dot", color="#00BCD4")

    def calculate(self, df: pd.DataFrame) -> None:
        high = self._col(df, "high")
        low = self._col(df, "low")
        step = self.parameters["acceleration"]
        max_step = self.parameters["maximum"]

        self.lines["sar"].data = talib.SAR(high, low, acceleration=step, maximum=max_step)
