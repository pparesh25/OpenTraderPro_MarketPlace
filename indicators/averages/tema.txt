# plugin-type: indicator
"""Triple Exponential Moving Average indicator."""

import pandas as pd
import talib

from opentrader.indicators.base import BaseIndicator


class TEMA(BaseIndicator):
    """Triple Exponential Moving Average."""

    name = "Triple Exponential Moving Average"
    short_name = "TEMA"
    description = "Triple smoothing for even less lag than DEMA."
    overlay_on_price = True

    def __init__(self) -> None:
        super().__init__()
        self.parameters = {"period": 20}
        self.add_line("tema", "TEMA", color="#4CAF50")

    def calculate(self, df: pd.DataFrame) -> None:
        period = self.parameters["period"]
        close = self._col(df, "close")
        self.lines["tema"].data = talib.TEMA(close, timeperiod=period)
