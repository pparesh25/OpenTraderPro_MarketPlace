# plugin-type: indicator
"""Exponential Moving Average indicator."""

import pandas as pd
import talib

from opentrader.indicators.base import BaseIndicator


class EMA(BaseIndicator):
    """Exponential Moving Average — weighted toward recent prices."""

    name = "Exponential Moving Average"
    short_name = "EMA"
    description = "Exponentially weighted average of closing prices."
    overlay_on_price = True

    def __init__(self) -> None:
        super().__init__()
        self.parameters = {"period": 20}
        self.add_line("ema", "EMA", color="#FF9800")

    def calculate(self, df: pd.DataFrame) -> None:
        period = self.parameters["period"]
        close = self._col(df, "close")
        self.lines["ema"].data = talib.EMA(close, timeperiod=period)
