# plugin-type: indicator
"""Money Flow Index indicator."""

import numpy as np
import pandas as pd
import talib

from opentrader.indicators.base import BaseIndicator


class MFI(BaseIndicator):
    """MFI — volume-weighted RSI."""

    name = "MFI"
    short_name = "MFI"
    description = "Money Flow Index — volume-weighted momentum."
    overlay_on_price = False

    def __init__(self) -> None:
        super().__init__()
        self.parameters = {"period": 14, "overbought": 80, "oversold": 20}
        self.add_line("mfi", "MFI", color="#2962FF")
        self.add_line("overbought", "Overbought", color="#FF5252", style="dash")
        self.add_line("oversold", "Oversold", color="#4CAF50", style="dash")

    def calculate(self, df: pd.DataFrame) -> None:
        period = self.parameters["period"]
        high = self._col(df, "high")
        low = self._col(df, "low")
        close = self._col(df, "close")
        volume = self._col(df, "volume")

        self.lines["mfi"].data = talib.MFI(high, low, close, volume, timeperiod=period)
        
        n = len(df)
        self.lines["overbought"].data = np.full(n, self.parameters["overbought"])
        self.lines["oversold"].data = np.full(n, self.parameters["oversold"])

    def get_value_range(self):
        return (-5, 105)
